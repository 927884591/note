# 使用

使用4的版本和以前的3版本有些不同，跟vue3一样，vuerouter也是用来compositionAPI。

//router/index.js文件
import {createRouter,createWebHistory,createWebHashHistory} from "vue-router" //导入了三个函数，一个是床见router的函数和history模式，hash模式，这三个都是函数
import Home from "home"	//路径随便乱写了，自行更改
const routers = [
  {
    //写前端路由的影射关系 
    path:"/home",
  	component: Home 
  }
]
const router = createRouter({
  routers,
 	history:createWebHistory() //使用history的模式
  })
export default router

# 路由可以使用name跳转

每个组件都可以使用name的变量，然后在router表中使用name属性跳转

# 动态路由

如果需要setup中拿到router中动态的name，则需要导入vue-router中提供的api，vuerouter专门给setup提供userRouter这个API方便拿到动态的name

const routes = [
  {
    path: "/",
    name: "Home",
    component: Home,
  },
  {
    path: "/user/:name", //动态路由匹配，url中user/后必须带油参数才能匹配的上
    component: import("../views/User.vue"), //路由懒加载
  },
];


<router-link to="/user/kobe"><router-link> //这时候我们给user这个组件传过去一个name参数kobe

# 缺省路由

如果没有匹配到url的路径则会匹配我们相对应的组件显示

const routers = [
  {
    path:"/",
    redicret:Home 
  },
  {
    path:"/:pathMatch(.*)",   //缺省路由匹配，如果没有相对应的url匹配上则会匹配这个路由

# 动态添加路由

动态添加路由，可以应用于权限管理，可以先判断是否是管理员，然后动态添加路由

import {createRouter,createWebHistory} from "vue-router"
const routers = [
  {
    path:"/home",
    component:()=>import("./home.vue")
  }
]
const router = createRouter({
  routers,
  history:createWebHistory()
})
if(管理员){ //判断如果是管理员的话动态添加路由
  router.addRouter({
    path:"/category",
    component:()=>import("./category.vue")
  })
   
}

# 给二级路由添加动态路由

import {createRouter,createWebHistory} from "vue-router"
const routers = [
  {
    path:"/home",
    name:"home" //如果需要给二级及以上的路由添加动态路由的话给个name的属性可以给指定的路由添加路由
    component:()=>import("./home.vue"),
    children:[
      {
        path:"/message",
        name:"message"   
        component:()=>import("./message.vue")
      }
    ]
  }
]
const router = createRouter({ //顶层路由
  routers,
  history:createWebHistory()
})


router.addRouter("home",{
  path:"/moment",
  component:()=>import("./moment.vue")
})

# 路由导航守卫

导航守卫就是需要跳转时，拦截，然后进行一些操作。例如：用户如果想查看个人信息，但用户没有登录，我们这时候不应该给他跳转到个人信息的路由中，而是跳转到登录页面中。

# 全局导航守卫

import Longin from "./Longin.vue"
